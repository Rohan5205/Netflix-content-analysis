def catigorize_col(data,col,labels):
    edges=[data[col].describe()['min'],
           data[col].describe()['25%'],
           data[col].describe()['50%'],
           data[col].describe()['75%'],
           data[col].describe()['max']]
    data[col]=pd.cut(data[col],edges,labels=labels,duplicates='drop')
    return data

genre_count = data.groupby('Title')['Genre'].count().reset_index()
genre_count.columns = ['Title', 'Genre_Count']

popularity_data = data[['Title', 'Popularity']].drop_duplicates()
genre_pop = pd.merge(genre_count, popularity_data, on='Title')

single_genre_avg = genre_pop[genre_pop['Genre_Count'] == 1]['Popularity'].mean()
multi_genre_avg = genre_pop[genre_pop['Genre_Count'] > 1]['Popularity'].mean()

print(f"Average Popularity - Single Genre: {single_genre_avg:.2f}")
